Sample Python Code - Recommendation System

import pandas as pd
from surprise import Dataset, Reader, KNNBasic

ratings_dict = {
    "itemID": [1, 1, 1, 2, 2, 3],
    "userID": [9, 32, 2, 45, 32, 9],
    "rating": [3, 2, 4, 5, 3, 4]
}
df = pd.DataFrame(ratings_dict)

reader = Reader(rating_scale=(1, 5))
data = Dataset.load_from_df(df[['userID', 'itemID', 'rating']], reader)

algo = KNNBasic()
trainset = data.build_full_trainset()
algo.fit(trainset)

pred = algo.predict(9, 2)
print(f"Predicted rating for user 9 on item 2: {pred.est}")

Artifact 4: Data Challenge Scenarios – Addressing Quality, Privacy, and Bias in Machine Learning

Objective

Demonstrate the ability to identify and solve real-world data challenges affecting AI model accuracy, fairness, and compliance.

Scenario 1: Data Quality

  • Challenges: Missing data, duplicates, outliers
  • Solutions: Imputation, normalization, outlier removal
  • Tools: Python (pandas, numpy, sklearn)
import pandas as pd
import numpy as np
from scipy import stats
from sklearn.preprocessing import MinMaxScaler

data = pd.read_csv('financial_data.csv')
data['income'].fillna(data['income'].mean(), inplace=True)
data['occupation'].fillna(data['occupation'].mode()[0], inplace=True)
data.drop_duplicates(inplace=True)
z_scores = np.abs(stats.zscore(data.select_dtypes(include=[np.number])))
data = data[(z_scores < 3).all(axis=1)]
scaler = MinMaxScaler()
data[['income', 'age']] = scaler.fit_transform(data[['income', 'age']])

Scenario 2: Privacy & Security

  • Challenges: Protecting sensitive healthcare data under GDPR and HIPAA
  • Solutions: Data anonymization, AES encryption, role-based access
  • Tools: Python cryptography library
from cryptography.fernet import Fernet

key = Fernet.generate_key()
cipher_suite = Fernet(key)

sensitive_data = b"Patient PHI data"
encrypted_data = cipher_suite.encrypt(sensitive_data)
decrypted_data = cipher_suite.decrypt(encrypted_data)
print(decrypted_data.decode())

Scenario 3: Bias and Fairness

  • Challenges: Class imbalance and bias in hiring datasets
  • Solutions: SMOTE oversampling, bias auditing, threshold adjustments
  • Tools: imbalanced-learn, AIF360
from imblearn.over_sampling import SMOTE
from collections import Counter

X = data.drop('target', axis=1)
y = data['target']

print('Original dataset shape %s' % Counter(y))
sm = SMOTE(random_state=42)
X_res, y_res = sm.fit_resample(X, y)
print('Resampled dataset shape %s' % Counter(y_res))
